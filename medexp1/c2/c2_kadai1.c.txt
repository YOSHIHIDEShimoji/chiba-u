float *movingAverage(float *signal, int length, int K)
{
    /* NULLの代わりにmallocで動的配列を宣言すること*/
    float* averagedSignal = NULL;

    for (int i=0; i<length; i++)
    {
        float sum = 0.0;
        float count = 0.0;

        for (int j=i-K; j<=i+K; j++)
        {
            /* 条件式を設定すること */
            if (jが配列の範囲内かどうかの判定)
            {
                sum += signal[j];
                count++;
            }    
        }
        averagedSignal[i] = sum / count;
    }

    return averagedSignal;
}
